
\chapter{Review of Complex number $\mathbb{C}$}
The goal of this chapter is to review what we have seen in Analysis I in order to be more comfortable when introducing new materials. As an addition I will add a bit of context on where and why complex analysis differs from multivariate analysis (this is not in the course but is always nice to have in order to better understand the materials).


\begin{parag}{Imaginary unit}
    As seen in previous classes, we define $i$ the \important{imaginary unit} such that:
	\begin{align*} i^2 = -1 \end{align*}
	We also define $z \in \mathbb{C}$ as:
	\begin{align*} z = x + iy \mathspace \mathspace x, y \in \mathbb{R} \end{align*}
	Furthermore:
	\begin{align*} 
		x &= \text{Re}z \mathspace \text{real part}\\
		y &= \text{Im}z \mathspace \text{imaginary part}
	\end{align*}
\end{parag}
The field $\mathbb{C}$ can by fully mapped to $\mathbb{R}^2$, what this means is that every element in $\mathbb{C}$ can be directly '\textit{translated}' to $ \mathbb{R}^2$ \important{and} vis versa.

\begin{parag}{Complex Conjugate}
    \begin{definition}
    Let us define the complex conjugate of $z$ as:
	\begin{align*} \overline{z} = x - iy  \end{align*}
    \end{definition}
\end{parag}
\begin{parag}{Absolute Value}
    \begin{definition}
    The absolute value (modulus) is defined as:
	\begin{align*} | z | = \sqrt{x^2 + y^2} \end{align*}
    \end{definition}
	\begin{subparag}{Intuition}
	    The reason for this definition really comes from Pythagoras theorem and the $L_2$ norm in $\mathbb{R}^2$.
	\end{subparag}
	Furthermore $| z_0 - z_1| = d\left(z_0, z_1\right)$ (the distance between those two number).\\
	We can define domain in $\mathbb{C}$ as:
	\begin{align*} Q = \{z \in \mathbb{C}: |z - \left(1 + i\right)| < 1\} \end{align*}
	Which is the disk centred in $z = 1 + i$ of radius $1$.
\end{parag}

\begin{parag}{Euler's Formula}
    \begin{definition}
    For $\theta \in \mathbb{R}$:
	\begin{align*} e^{i\theta} = \cos \theta + i \sin \theta  \end{align*}
    \end{definition}
	So $e^{i\theta}$ lies on the unit circle around $z = 0$ (because $|e^{i\theta}| = 1$)
	\begin{definition}
	The Complex conjugate of the Euler formula is defined as
	\begin{align*} \overline{e^{i\theta}} &= \overline{\cos \theta + i\s\in\theta} \\
		&= \cos \theta - i \sin \theta\\
		&= \cos(-\theta) + i \sin\left(-\theta\right)\\
		&= e^{-i\theta}
	\end{align*}
	\end{definition}
	\begin{subparag}{Geometric intuition}
	    What happens when doing the complex conjugate (if we see $\mathbb{C}$ as a plan in $ \mathbb{R}^2$) is a symmetry around the $x$ axis. \\
		$ $\\
		And this totally makes send when we recall on how it was defined when $z =  x + iy$, $\overline{z} = x - iy$ we map the $\text{Im}$ part to his opposite.
	\end{subparag}
\end{parag}
\begin{framedremark}
	The rigorous definition of $e^{z}$ is by the Taylor series. One should more see it as $\exp$ being \important{defined} by the Taylor serie and not the other way around. There is a great video about the definition of the exponential by 3Blue1Brown about taking the exponent of a matrix.
\end{framedremark}





\begin{parag}{Algebraic Operations}
    Let $z_1 = x_1 + iy_1$ and $z_2 = x_2 + iy_2$ we have the followings:
		 \begin{align*} 
				z_1 + z_2 &= \left(x_1 + iy_1\right) + \left(x_2 + iy_2\right)\\
						  &= \left(x_1 + x_2\right) + i \left(y_1 + y_2\right)
			\end{align*}
		 \begin{align*} 
				z_1 - z_2 = \left(x_1 - x_2\right) + i \left(y_1 - y_2\right)
			\end{align*}
		\begin{align*} 
			z_1 \cdot z_2 &= (x_1 + iy_1) \cdot \left(x_2 + iy_2\right)\\
						  &= x_1x_2 - y_1y_2 + i\left( y_1x_2 + x_1y_2\right)
		\end{align*}
	\begin{align*} 
		z \cdot  \overline{z} = x^2 + y^2 = |z|^2 \in \mathbb{R}
	\end{align*}
		\begin{align*} 
			\frac{z_1}{z_2} &= \frac{\overline{z_2}z_1}{\overline{z_2}z_2}\\
							&= \frac{\overline{z_2}z_1}{|z_2|^2} \mathspace \text{ if } z_2 \neq 0
		\end{align*}
	
\end{parag}



\begin{parag}{Polar coordinates}
	\begin{itemize}
		\item If $z \in \mathbb{C}$ then $z =  |z|e^{i\theta}$ for some $\theta \in \mathbb{R}$.
		\item This also means that $z^{k} =  |z|^{k}e^{ik\theta}$ for $k \in \mathbb{Z}$
	\end{itemize}

	Let us define now our two complex numbers in polar coordinates $z_1 = |z_1|e^{i\theta_1}, z_2 = |z_2|e^{i\theta_2}$. If we want to apply algebraic operations defined previously to our new notation we would have:
	\begin{align*} z_1 \cdot  z_2 &= |z_1||z_2| e^{i\left(\theta_1 + \theta_2\right)} \end{align*}
	Dividing our two numbers really comes down to multiply by the inverse which is defined as $z_2^{-1} = \frac{1}{|z_2|}e^{-i\theta_2}$

	Lastly:
	\begin{align*} z_1 = z_2 \iff |z_1| = |z_2| \land \theta_1 - \theta_2 =  2\pi \cdot  k, \mathspace k \in \mathbb{Z}  \end{align*}
	\begin{subparag}{Geometric Intuition}
	    Multiplication by $e^{i\theta_0}$ corresponds to a rotation by an angle $\theta_0$.
	\end{subparag}
\end{parag}

\begin{framedremark}\label{rem:visualization}

\begin{parag}{Example}
    Let us try to solve the following equation:
	\begin{align*} z^5 = 2 \end{align*} 
	To do so let us write our number $z$ into polar coordinate: $z = |z|e^{i\theta}$. If we insert our number into our equation we have:
	\begin{align*} 
		2 &= |z|^{5}e^{i5\theta}
	\end{align*}
	As $|e^{i5\theta}| = 1$ we have that 
	\begin{align*} 2 = |z|^{5} \implies |z| = 2^{\frac{1}{5}} \end{align*}
	And as $2$ is a real number this implies that $5\theta = 2\pi k \implies \theta =  \frac{2\pi k}{5}$ where $k \in \mathbb{Z}$ and if we want to take the principal arguments, we have as solution:
	\begin{align*} \theta \in \{-\frac{4\pi}{5}, -\frac{2\pi}{5}, 0, \frac{2\pi}{5}, \frac{4\pi}{5}\} \end{align*}
	More generally, a $k$-th order equation will gives us $k$ solutions.
\end{parag}

\section{Construction of $\mathbb{C}$ (not in the course)}
The goal of this section is to add a bit more context on the $\mathbb{C}$ field. First let us recall how real analysis worked. We used the field $\left(\mathbb{R}, +, \cdot \right)$ which is an \important{ordered} field, i.e. we have an ordering on our elements. Formally:
\begin{definition}{Ordered field}
	$ $\\
	\begin{itemize}
		\item $\forall x, y \in \mathbb{R} $ precisely one of $x < y,  x > y, x = y$ holds true.
		\item $x, y > 0 \implies x \cdot  y > 0$
	\end{itemize}
	This is the axiomatic definition of an ordered field.
\end{definition}
And as we already know, $\mathbb{C}$ is not an ordered field, $2 + i$ cannot be greater nor smaller to $12$.\\
$ $\\
One of the reason on why $\mathbb{C}$ has come up is that the field $\left(\mathbb{R}, +, \cdot \right)$ is \important{not algebraically closed}. We can find polynomials with real coefficient which don't have real roots. $x^2 + 1$ has not root over $ \mathbb{R}$. This is a major issue, we need to find a field which is algebraically closed. The question now that was asked: does $\left(\mathbb{R}^2, +\right)$ (which is already an abelian group) can be turned into a field $\left( \mathbb{R}^{2}, +, ?\right)$? And what about $\left(\mathbb{R}^{n}, +\right)$ is there a way to 'create' a multiplication which would be suitable?\\
$ $\\
For $n = 1$ this is just the real number. For $n = 2$ the resulting field you will get is $\mathbb{C}$. For $n =  4$ the answer is the Hamilton group (kind of) and the same for $n = 8$. For all other number it is just \important{impossible}! And this is pretty bizarre, it worked for $1, 2, 4, 8$ and then it stopped.
\begin{theoreme}
	If we can create a field on $ \mathbb{R}^2$ then 
	\begin{align*} \exists z \in\left(\mathbb{R}^{2}, +, \cdot \right) \text{ s.t. } z^2 =  -1 \end{align*}
\end{theoreme}
\begin{proof}

    Let us choose a basis element $\left\{1, e\right\}$ of $\mathbb{R}^{2}$. This directly implies that:
	\begin{align*} 
		z = x \cdot 1 + y \cdot e \text{ , for some } x, y \in \mathbb{R}
	\end{align*}
	If we want to multiply our number by itself:
	\begin{align*} 
		z^2 &= x^2 \cdot 1 + 2xy \cdot e + y^2 \cdot  e^2
	\end{align*}
	\begin{framedremark}
	A strange thing we did here is to add the $2xy \cdot e$. Recall that previously, when we had a vector in $\mathbb{R}^{2}, x$ that we squared, the output that we will get would be:
	\begin{align*} 
		\begin{pmatrix} x_1 \\ x_2 \end{pmatrix}  \cdot \begin{pmatrix} x_1 \\ x_2 \end{pmatrix}  = x_1^2 + x_2^2
	\end{align*}
	Or if we have written them using the basis vector $e_1 = \begin{pmatrix} 1 \\ 0 \end{pmatrix} , e_2 =  \begin{pmatrix} 0 \\ 1 \end{pmatrix} $ this would gives us:
	\begin{align*} 
		\left(x_1e_1 + x_2e_2\right) \cdot \left(x_1e_1 + x_2e_2\right)
	\end{align*}
	The important thing is that both equation above are \important{totally equivalent}. If we develop this we will get:
	\begin{align*} 
		x_1^2 e_1^2 + 2x_1x_2e_1e_2 +x_2^2e_2^2
	\end{align*}
	And because $e_1$ and $e_2$ are orthogonal, by definition $e_1e_2 = 0$:
	\begin{align*} 
		= x_1^2e_1^2 + x_2^2e_2^2 = x_1^2 + x_2^2
	\end{align*}
	\end{framedremark}
	Now we want to rewrite $e^2$ using the basis that we introduced before:
	\begin{align*} 
		e^2 = a \cdot 1 + b \cdot e, \mathspace \mathspace a, b \in \mathbb{R}
	\end{align*}
	This implies that:
	\begin{align*} 
		z^2 =  \left(x^2 + ay^2\right) \cdot  1 + \left(2xy + by^2\right) \cdot  e
	\end{align*}
	Now we want to show that there exists the pair $\left(x, y\right) \in \mathbb{R}^{2}$ such that $z^2 = -1$. Then we must have the following:
	\begin{align*} 
		2xy + by^2 = 0 \iff \begin{cases}
		y =  0 \\ x = -\frac{b}{2}y
		\end{cases} \text{ or}
	\end{align*}
	But here $y= 0$ is not sufficient because this implies for the second relation:
	\begin{align*} 
		x^2 + ay^2 = -1
 	\end{align*}
	that $x^2 = -1$ where $x \in \mathbb{R}$ which is impossible. We therefore look into the second case, $y \neq 0$ and $x = -\frac{b}{2}y$. In this case this implies that:
	\begin{align*} 
		z^2 &= \left(-\frac{b}{2}y\right)^2 + ay^2 \cdot 1 + \underbrace{\left(2\left(\frac{-b}{2}\right)y^2 + by^2\right)\cdot  e}_{= 0}\\
		&= \left(a + \frac{b^2}{4}\right)y^2 \cdot  1
	\end{align*}
	The second claim is that $a + \frac{b^2}{4} < 0$
		\begin{proof}
		    
		
	    The proof will be done by contradiction. Suppose that $a + \frac{b^2}{4} \geq 0$. This implies that we can take the square root (in the sense of $ \mathbb{R}$) and therefore we have that:
		\begin{align*} 
			z =  \pm y\sqrt{a + \frac{b^2}{4}}
		\end{align*}
		Which implies that:
		\begin{align*} 
			\left(z - y\sqrt{a + \frac{b^2}{4}}\right)\left(z + y\sqrt{a + \frac{b^2}{4}}\right) = 0
		\end{align*}
		This implies that either $z = y \sqrt{a + \frac{b^2}{4}} = x \cdot 1 + y \cdot  e$ or the same with a negative sign.
		\begin{align*} 
			\pm y\sqrt{a + \frac{b^2}{4}}  \cdot  1=  x \cdot  1 + y \cdot  e
		\end{align*}
		Since $\left\{1, e\right\}$ are linearly independent, this equality is impossible unless $y = 0$ which contradicts the case we are studying ($y \neq 0$). Therefore $a + \frac{b^2}{4} < 0$
		\end{proof}
		Now our goal is to proof there exists a $y$ such that $\left(a + \frac{b^2}{4}\right)y^2 = -1$. We can set
		\begin{align*} 
			y := \frac{1}{-\sqrt{a + \frac{b^2}{4}}}, \mathspace \mathspace x = \frac{-b}{2}y
		\end{align*}
		This directly implies that:
		\begin{align*} 
			z^2 &= \left(a + \frac{b^2}{4}y^2\right)\\
				&= \left(a + \frac{b^2}{4}\right)\left(\frac{-1}{a + \frac{b^2}{4}}\right) = -1
		\end{align*}

\end{proof}
The claim we just prove is not trivial at all, if we want to turn $\mathbb{R}^{2}$ into a field, then there has to be an element of the field such that $d = -1$. This also implies that our multiplication will be a new one which will be pretty counter-intuitive for us. Now let us pick our element $z \in \left(\mathbb{R}^{2}, + , \cdot \right)$ such that $z^2 = -1$, and let us call it \textbf{i}.\\
$ $\\
Now we can define our field $\mathbb{C}$:
\begin{definition}

	Let us define the \important{complex field} $\mathbb{C}$:
	\begin{align*} 
		\mathbb{C} &:= \mathbb{R}[i] := \left\{a + bi, \mathspace a, b \in \mathbb{R}\right\}\\
				   &=  \mathbb{R}[X]/(X^2+1) := \left\{a + bX \in \mathbb{R}[X]\right\}\\
	\end{align*}
\end{definition}
	$\mathbb{C}$ is algebraically closed

\begin{theoreme}
	Any $P \in \mathbb{C}\left[X\right]$ non-zero (not the zero polynomial) has at least a root in $\mathbb{C}$.
\end{theoreme}
\begin{proof}
    To prove this we will do it by contradiction. Assume that $P$ is a non-constant polynomial and suppose it \important{has no root} in $\mathbb{C}$. Then this implies that:
	\begin{align*} 
		P\left(z\right) \neq 0 \mathspace \forall z \in \mathbb{C}
	\end{align*}
	Therefore, we can define:
	\begin{align*} 
		f\left(z\right) = \frac{1}{P\left(z\right)}
	\end{align*}
	Since $P\left(z\right) \neq 0$ everywhere and $P$ is holomorphic, $f$ is also holomorphic on all of $\mathbb{C}$ (entire).\\
	$ $\\
	The next step of our proof is to show that $f$ is bounded. Let 
	\begin{align*} 
		P\left(z\right) = a_nz^{n} + a_{n-1}z^{n-1} + \cdots + a_0
	\end{align*}
	For large $\left\| z\right\|$, the leading term dominates:
	\begin{align*} 
		\left\|P\left(z\right)\right\|  \approx \left\|a_n\right\|\left\|z\right\|^{n}
	\end{align*}
	More precisely, there exists $R$ such that if $\left\|z\right\| > R$ then:
	\begin{align*} \left\|P\left(z\right)\right\| \geq \frac{\left\|a_n\right\|}{2}\left\|z\right\|^{n} \end{align*}
	Thus:
	\begin{align*} 
		\left\|f\left(z\right)\right\| =  \frac{1}{\left\|P\left(z\right)\right\|} \leq \frac{2}{\left\|a_n\right\|\left\|z\right\|^{n}}
	\end{align*}
	Therefore, when $\left\|z\right\| \to \infty$ we have that
	\begin{align*} 
		f\left(z\right) \to 0
	\end{align*}
	Hence $f$ is bounded outside a large disk. And inside $\left\|z\right\| < R$, the continuous function $f$ on a compact set is also \important{bounded}. Therefore, $f$ is bounded on all of $\mathbb{C}$.\\
	$ $\\
	All the work is done now, all we have to do is to apply Liouville's theorem:
	\begin{theoreme}
		Every bounded entire function is constant
	\end{theoreme}
	Which therefore implies that $f$ must be constant. But we just said
	\begin{align*} 
		f\left(z\right) = \frac{1}{P\left(z\right)}
	\end{align*}
	which means that $P$ is constant which is a contradiction.\\
	$ $\\
	Conclusion:
	\begin{center}
		Every non-constant polynomial in $\mathbb{C}\left[x\right]$ has a complex root.
	\end{center}
\end{proof}



\chapter{Complex Functions}
\begin{definition}
	We call $f$ a complex function as:
	\begin{align*} 
		f : \mathbb{C} \to \mathbb{C} \\
		z \mapsto f(z) &= f(x + iy)\\
					   &= u(x, y) + iv\left(x, y\right)
	\end{align*}
	In particular we have $u, v : \mathbb{R}^2 \to \mathbb{R}$
\end{definition}
A nice thing to see here is the relation between $\mathbb{R}^2$ and $\mathbb{C}$. As said before we can always map a complex number to a number in $ \mathbb{R}^2$. By 'splitting' our function into to  two sub functions $u, v$ which map to $ \mathbb{R}$. We effectively transformed our function $f : \mathbb{C} \to \mathbb{C}$ into two sub function that goes from $ \mathbb{R}^2 \to \mathbb{R}$ which therefore point to $ \mathbb{R}^2$ (if we replaced the 1 by a unit vector, and $i$ as another unit vector).


\begin{parag}{Examples}
    First let us see the simplest example, the \important{unit} function: $f\left(z\right)=  z =  x + iy$. In this case we can easily see that 
	\begin{align*} u(x, y) = x \mathspace \mathspace v\left(x, y\right) =  y \end{align*}
	A bit more sophisticated function could be for instance: $f\left(z\right) =  \overline{z} = x - iy$. Which gives us:
	\begin{align*} 
		u\left(x, y\right) =  x \mathspace \mathspace v\left(x, y\right) =  -y
	\end{align*}
	If we take the square we have $f\left(z\right) =  z^2 =  \left(x + iy\right)^2 = \left(x^2 - y^2\right) + i2xy$\\

	In all of those expression we can easily see where our function $u$ and $v$ are. But is it always the case?
\end{parag}

	\begin{align*} \sin z = \frac{e^{iz} - e^{-iz}}{2i} \end{align*}

	\end{framedremark}

	\end{align*}
	\begin{align*}

	\end{align*}

\section{Limit and Continuity}

\begin{definition}
We say $f: \mathbb{C} \to \mathbb{C}$ is \important{continuous} at a point $z_0 \in \mathbb{C}$ if $\forall \epsilon > 0 $ there exists $\delta > 0 $ such that $\forall z \in \mathbb{C}$ with $\left\|z - z_0\right\| \leq \delta$ we have:
\begin{align*} \left\|f\left(z\right) - f\left(z_0\right)\right\| \leq \epsilon \end{align*}
Equivalently:
\begin{align*} \lim_{z \to z_0} f\left(z\right) =  f\left(z_0\right) \end{align*}
\end{definition} 
\begin{framedremark}
This definition is directly mapped to the one seen in Analysis II. What we are really doing here in fact is the limit of our function the same way as we would have done in $ \mathbb{R}^2$. If we develop this:\\
    Let $f\left(z\right) = f\left(x + iy\right) =  u\left(x, y\right) + iv\left(x, y\right)$ then $f$ is continuous at $z_0 = x_0 + y_0i$ if and only if $u$ and $v$ are continuous at $\left(x_0, y_0\right)$ 
\end{framedremark}
\begin{parag}{Operation on Continuous Functions}
    If $f, g$ are continuous functions then the same is true for:
	\begin{itemize}
		\item $f+ g$
		\item $f \cdot  g$
		\item $\overline{f}$
		\item $f \circ g$
		\item $\vdots$
	\end{itemize}
	
\end{parag}


\subsection{Complex Derivative}

\begin{definition}
	We say $f : \mathbb{C} \to \mathbb{C}$ is \important{complex differentiable}  at a point $z_0 \in \mathbb{C}$ if:
	\begin{align*} \lim_{z \to z_0} \frac{f\left(z\right) - f\left(z_0\right)}{z - z_0} \text{ exists} \end{align*}
	We denote this limit by $f'\left(z_0\right)$
\end{definition}
\begin{definition}\label{def:holomorphic}
If $f$ is differentiable at every point of an open set $\omega \subseteq \mathbb{C} $, we say $f$ is \important{holomorphic} on $\omega$.
\end{definition}
\begin{framedremark}
	In many books about complex analysis we use the term analytic instead of holomorphic, both terms are equivalent for function of complex variable, but this equivalence is a deep theorem and not a definition. The definition~\ref{def:holomorphic} here is the definition of \important{holomorphic}. On the other hand, analytic on $\omega$ means that around every point $z_0 \in \omega$, $f$ can be represented by a convergent power series:
\begin{align*} 
	f\left(z\right) = \sum_{n= 0}^{\infty}a_n\left(z - z_0\right)^{n}
\end{align*}
In some neighbourhood of $z_0$. This is an \textit{a priori} much stronger condition.\\
$ $\\
The fact that $f$ is holomorphic $\iff$ $f$ is analytic (on $\omega$) can be proven, and we will be able to do prove it.\\
$ $\\
The Analytic $\implies$ holomorphic proof is pretty straightforward: as $f$ can be represented by a power series and that a power series is differentiable (within their radius of convergence) then $f$ is also differentiable the same way.\\
$ $\\
For the holomorphic $\implies $ Analytic proof. This is harder and we will go through it later when we will have introduced Cauchy's integral formula\\
$ $\\
This theorem doesn't work in $ \mathbb{R}$ a function can be infinitely differentiable (smooth) on $ \mathbb{R}$ without being analytic. For instance:
\begin{align*} 
	f\left(x\right) = \begin{cases}
		e^{-\frac{1}{x^2}} \mathspace \mathspace &x \neq 0\\
		0 &x = 0
	\end{cases}
\end{align*}
This is $C^{\infty}$ everywhere, but its Taylor series at $0$ is identically $0$, so it fails to represent $f$ near $0$
\end{framedremark}
\begin{definition}
If $f$ is holomorphic on $\mathbb{C}$, we call f \important{entire}.
\end{definition}


\begin{framedremark}
The same rules of differentiation of products (product rule, Leibniz rule), Compositions (chain rule), etc. are as in the real case.
\end{framedremark}

\paragraph{Recall on topology}%
\label{par:Recall on topology}
\begin{parag}{Open set}
	\begin{definition}
	$\omega \subseteq \mathbb{C}$ is \important{open} if for all $z_0 \in \omega$, there exists $\Gamma > 0$ such that the disk $D_\Gamma\left(z_0\right) =  \left\{z \in \mathbb{C}: \left\|z - z_0\right\| < r\right\} \subseteq \omega$
	\end{definition}
    \begin{subparag}{Examples}
        \begin{itemize}
			\item $\mathbb{C}$ is open
			\item $\mathbb{C}^{\ast} =  \mathbb{C}\setminus \left\{0\right\}$ is open
			\item $\left\{z \in \mathbb{C} : \text{Re }z \geq 0\right\}$ is \textcolor{red}{not} open 
        \end{itemize}
    \end{subparag}
\end{parag}




\begin{parag}{Examples of Differentiation}
    Let $f\left(z\right) = z$ our goal is to compute the derivative of $f$ for all $z_0 \in \mathbb{C}$. By the definition:
	\begin{align*} 
		f'\left(z_0\right) =  \lim_{z \to z_0} \frac{z - z_0}{z - z_0} =  1
	\end{align*}
	In particular, $f$ is (holomorphic and) entire.

	Let us try a little more sophisticated example. Let $f\left(z\right) =  \frac{1}{z}$. For all $z_0 \in \mathbb{C}^{\ast}$:
	\begin{align*} 
		f'\left(z_0\right) = -\frac{1}{z_0^2}
	\end{align*}
	So $f$ is holomorphic on $\mathbb{C}^{\ast}$ but not entire.
\end{parag}
